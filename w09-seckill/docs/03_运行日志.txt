========================================
  秒杀系统 — 全链路运行日志
========================================

=== 系统架构 ===

  ┌─────────────────────────────────────────────────────────────┐
  │                     用户秒杀请求                             │
  └────────────────────────────┬──────────────────────────────────┘
                               ↓
  ┌────────────────────────────────────────────────────────────┐
  │  Gateway (8090) — Sentinel 限流 + 路由分发                 │
  └────────────────────────────┬───────────────────────────────┘
                               ↓
  ┌────────────────────────────────────────────────────────────┐
  │  ① Seckill Service (8084) — 频率限制 + 布隆过滤 + Lua 扣库存│
  └────────────────────────────┬───────────────────────────────┘
                               ↓ Kafka: seckill-order
  ┌────────────────────────────────────────────────────────────┐
  │  ② Order Service (8085) — 创建订单 + MySQL 扣库存          │
  └────────────────────────────┬───────────────────────────────┘
                               ↓ Kafka: seckill-payment
  ┌────────────────────────────────────────────────────────────┐
  │  ③ Payment Service (8086) — 模拟支付 + 回调通知            │
  └────────────────────────────┬───────────────────────────────┘
                               ↓ Kafka: payment-result
  ┌────────────────────────────────────────────────────────────┐
  │  ② Order Service (8085) — 更新订单状态为 PAID              │
  └────────────────────────────────────────────────────────────┘

========================================
  第一部分：环境准备 & 启动命令
========================================

=== 1. 前置中间件（需提前启动） ===

  MySQL  — localhost:3306
  Redis  — localhost:6379
  Kafka  — localhost:9092（含 ZooKeeper）
  Nacos  — localhost:8848（单机模式）

  验证命令:
  $ netstat -an | grep -E "LISTENING" | grep -E ":(3306|6379|9092|8848) "

  实际输出:
  TCP  0.0.0.0:3306  LISTENING ✓
  TCP  0.0.0.0:6379  LISTENING ✓
  TCP  0.0.0.0:8848  LISTENING ✓
  TCP  0.0.0.0:9092  LISTENING ✓

=== 2. 编译打包 ===

  $ cd D:\workspace\kungfu\w09-seckill

  # 安装父 POM 到本地仓库（只装 POM，不递归子模块）
  $ mvn install -N

  # 安装 common 模块到本地仓库（其他模块依赖它）
  $ mvn install -pl seckill-common -DskipTests

  # 全量打包
  $ mvn package -DskipTests

=== 3. 数据库初始化 ===

  命令（需要 MySQL connector jar 在 classpath）:
  $ java -cp "seckill-common\target\classes;%MYSQL_JAR%" ^
      com.kungfu.seckill.common.util.DatabaseInitializer

  实际输出:
  [OK] kungfu_seckill 数据库初始化完成!
    - t_seckill_activity  (1 条初始数据: iPhone 15 限时秒杀, 库存 100)
    - t_seckill_stock     (1 条初始数据: available_stock=100)
    - t_seckill_order     (空表)
    - t_seckill_payment   (空表)
    - undo_log            (Seata AT 模式)

=== 4. 启动微服务（每个服务开一个终端） ===

  # 终端1: 秒杀核心服务 (port 8084)
  $ mvn spring-boot:run -pl seckill-service

  # 终端2: 订单服务 (port 8085)
  $ mvn spring-boot:run -pl seckill-order

  # 终端3: 支付服务 (port 8086)
  $ mvn spring-boot:run -pl seckill-payment

  # 终端4: API 网关 (port 8090)
  $ mvn spring-boot:run -pl seckill-gateway

  验证端口:
  $ netstat -an | grep -E "LISTENING" | grep -E ":(8084|8085|8086|8090) "

  实际输出:
  TCP  0.0.0.0:8084  LISTENING ✓  (seckill-service)
  TCP  0.0.0.0:8085  LISTENING ✓  (seckill-order)
  TCP  0.0.0.0:8086  LISTENING ✓  (seckill-payment)
  TCP  0.0.0.0:8090  LISTENING ✓  (seckill-gateway)

  验证 Nacos 注册:
  $ curl -s "http://localhost:8848/nacos/v1/ns/service/list?pageNo=1&pageSize=10"

  实际输出:
  {"count":4,"doms":["seckill-gateway","seckill-payment","seckill-service","seckill-order"]}

========================================
  第五部分：一键测试脚本
========================================

  以下脚本可在 bash 中一键执行全部测试（需先完成第一部分的环境准备和服务启动）:

  #!/bin/bash
  # ============================================
  # w09-seckill 全链路测试脚本
  # 前置条件: 4 个中间件 + 4 个微服务已启动
  # ============================================

  GATEWAY=http://localhost:8090

  echo "=== 1. 查询活动详情 ==="
  curl -s $GATEWAY/api/seckill/activity/1 | python -m json.tool

  echo "=== 2. 活动预热 ==="
  curl -s -X POST $GATEWAY/api/seckill/activity/1/warmup | python -m json.tool

  echo "=== 3. 执行秒杀 (userId=10001) ==="
  RESULT=$(curl -s -X POST $GATEWAY/api/seckill/1 -H "X-User-Id: 10001")
  echo "$RESULT" | python -m json.tool
  ORDER_NO=$(echo "$RESULT" | python -c "import sys,json; print(json.load(sys.stdin)['data'])")

  echo "=== 4. 查询秒杀结果（等待3秒） ==="
  sleep 3
  curl -s $GATEWAY/api/seckill/result/$ORDER_NO | python -m json.tool

  echo "=== 5. 重复秒杀拦截 ==="
  curl -s -X POST $GATEWAY/api/seckill/1 -H "X-User-Id: 10001" | python -m json.tool

  echo "=== 6. 频率限制测试 ==="
  curl -s -X POST $GATEWAY/api/seckill/1 -H "X-User-Id: 10002" | python -m json.tool
  curl -s -X POST $GATEWAY/api/seckill/1 -H "X-User-Id: 10002" | python -m json.tool

  echo "=== 7. 多用户秒杀 ==="
  for uid in 10003 10004 10005; do
    curl -s -X POST $GATEWAY/api/seckill/1 -H "X-User-Id: $uid"
    echo ""
    sleep 1
  done

  echo "=== 8. 查询剩余库存 ==="
  curl -s $GATEWAY/api/seckill/activity/1 | \
    python -c "import sys,json; print('remainStock:', json.load(sys.stdin)['data']['remainStock'])"

  echo "=== 9. 并发秒杀（10用户） ==="
  for uid in $(seq 20001 20010); do
    curl -s -X POST $GATEWAY/api/seckill/1 -H "X-User-Id: $uid" -o /tmp/seckill_$uid.txt &
  done
  wait

  success=0; fail=0
  for uid in $(seq 20001 20010); do
    code=$(cat /tmp/seckill_$uid.txt | python -c "import sys,json; print(json.load(sys.stdin)['code'])" 2>/dev/null)
    if [ "$code" = "200" ]; then success=$((success+1)); else fail=$((fail+1)); fi
  done
  echo "并发结果 — 成功: $success, 失败: $fail"

  echo "=== 10. 并发订单支付状态（等待4秒） ==="
  sleep 4
  paid=0
  for uid in $(seq 20001 20010); do
    orderNo=$(cat /tmp/seckill_$uid.txt | python -c "import sys,json; print(json.load(sys.stdin).get('data',''))" 2>/dev/null)
    status=$(curl -s $GATEWAY/api/seckill/result/$orderNo | python -c "import sys,json; print(json.load(sys.stdin)['data']['status'])" 2>/dev/null)
    echo "  $orderNo → $status"
    if [ "$status" = "PAID" ]; then paid=$((paid+1)); fi
  done
  echo "已支付: $paid / 10"

  echo "=== 11. 库存一致性验证 ==="
  remain=$(curl -s $GATEWAY/api/seckill/activity/1 | python -c "import sys,json; print(json.load(sys.stdin)['data']['remainStock'])" 2>/dev/null)
  total_users=$((5 + 10))
  expected=$((100 - total_users))
  echo "Redis remainStock: $remain"
  echo "预期: $expected, 实际: $remain"
  if [ "$remain" = "$expected" ]; then echo "✓ 库存一致性验证通过"; else echo "✗ 库存不一致!"; fi

  echo "=== 12. 异常场景 ==="
  curl -s $GATEWAY/api/seckill/activity/999 | python -m json.tool
  curl -s $GATEWAY/api/seckill/result/SK_NOT_EXIST | python -m json.tool

  echo "=== 测试完成 ==="
