## 二、部署步骤

### 2.1 基础环境准备

#### 所有机器执行

```bash
# 1. 更新系统
yum update -y

# 2. 安装 Docker
curl -fsSL https://get.docker.com | bash
systemctl start docker
systemctl enable docker

# 3. 配置时间同步
yum install -y chrony
systemctl start chronyd
systemctl enable chronyd

# 4. 配置主机名和 hosts
hostnamectl set-hostname <hostname>
cat >> /etc/hosts <<EOF
192.168.1.10 nginx-1
192.168.1.11 nginx-2
192.168.1.20 gateway-1
192.168.1.21 gateway-2
192.168.1.22 gateway-3
# ... 其他机器
EOF

# 5. 关闭防火墙（或配置规则）
systemctl stop firewalld
systemctl disable firewalld

# 6. 配置内核参数
cat >> /etc/sysctl.conf <<EOF
net.ipv4.tcp_tw_reuse = 1
net.ipv4.tcp_fin_timeout = 30
net.ipv4.tcp_keepalive_time = 1200
net.core.somaxconn = 65535
net.ipv4.ip_local_port_range = 1024 65535
EOF
sysctl -p
```

### 2.2 部署 Nacos 集群

#### 机器：192.168.1.60-62

```bash
# 1. 下载 Nacos

wget https://github.com/alibaba/nacos/releases/download/2.2.3/nacos-server-2.2.3.tar.gz
tar -xzf nacos-server-2.2.3.tar.gz

# 2. 配置集群
cd nacos/conf
cp cluster.conf.example cluster.conf
cat > cluster.conf <<EOF
192.168.1.60:8848
192.168.1.61:8848
192.168.1.62:8848
EOF

# 3. 配置数据库（使用 MySQL）
vi application.properties
# 修改以下配置
spring.datasource.platform=mysql
db.num=1
db.url.0=jdbc:mysql://192.168.1.80:3306/nacos?characterEncoding=utf8
db.user.0=nacos
db.password.0=nacos123

# 4. 启动 Nacos
cd /opt/nacos/bin
./startup.sh

# 5. 验证
curl http://192.168.1.60:8848/nacos/
```

### 2.3 部署 Redis Cluster

#### 机器：192.168.1.70-75

```bash
# 1. 安装 Redis
yum install -y redis

# 2. 配置 Redis（每台机器）
vi /etc/redis.conf
# 修改以下配置
bind 0.0.0.0
port 6379
cluster-enabled yes
cluster-config-file nodes-6379.conf
cluster-node-timeout 5000
appendonly yes
maxmemory 8gb
maxmemory-policy allkeys-lru

# 3. 启动 Redis
systemctl start redis
systemctl enable redis

# 4. 创建集群（在任意一台机器执行）
redis-cli --cluster create \
  192.168.1.70:6379 \
  192.168.1.71:6379 \
  192.168.1.72:6379 \
  192.168.1.73:6379 \
  192.168.1.74:6379 \
  192.168.1.75:6379 \
  --cluster-replicas 1

# 5. 验证集群
redis-cli -c -h 192.168.1.70 cluster nodes
```

### 2.4 部署 MySQL 主从

#### 主库：192.168.1.80

```bash
# 1. 安装 MySQL
yum install -y mysql-server
systemctl start mysqld
systemctl enable mysqld

# 2. 配置主库
vi /etc/my.cnf
[mysqld]
server-id=1
log-bin=mysql-bin
binlog-format=ROW
max_binlog_size=100M
expire_logs_days=7

# 3. 重启 MySQL
systemctl restart mysqld

# 4. 创建复制用户
mysql -uroot -p
CREATE USER 'repl'@'%' IDENTIFIED BY 'repl123';
GRANT REPLICATION SLAVE ON *.* TO 'repl'@'%';
FLUSH PRIVILEGES;
SHOW MASTER STATUS;  # 记录 File 和 Position
```

#### 从库：192.168.1.81-82

```bash
# 1. 安装 MySQL
yum install -y mysql-server
systemctl start mysqld
systemctl enable mysqld

# 2. 配置从库
vi /etc/my.cnf
[mysqld]
server-id=2  # 从库 2 使用 server-id=3
read_only=1

# 3. 重启 MySQL
systemctl restart mysqld

# 4. 配置主从复制
mysql -uroot -p
CHANGE MASTER TO
  MASTER_HOST='192.168.1.80',
  MASTER_USER='repl',
  MASTER_PASSWORD='repl123',
  MASTER_LOG_FILE='mysql-bin.000001',  # 从主库获取
  MASTER_LOG_POS=154;                  # 从主库获取
START SLAVE;
SHOW SLAVE STATUS\G  # 验证
```

### 2.7 部署 Nginx

#### 机器：192.168.1.10-11

```bash
# 1. 安装 Nginx
yum install -y nginx

# 2. 配置 Nginx
cat > /etc/nginx/conf.d/seckill.conf <<EOF
upstream gateway {
    server 192.168.1.20:8090 weight=1 max_fails=3 fail_timeout=30s;
    server 192.168.1.21:8090 weight=1 max_fails=3 fail_timeout=30s;
    server 192.168.1.22:8090 weight=1 max_fails=3 fail_timeout=30s;
    keepalive 100;
}

# 限流配置
limit_req_zone \$binary_remote_addr zone=seckill:10m rate=100r/s;
limit_conn_zone \$binary_remote_addr zone=addr:10m;

server {
    listen 80;
    server_name seckill.example.com;
    
    access_log /var/log/nginx/seckill_access.log;
    error_log /var/log/nginx/seckill_error.log;
    
    # 限流
    limit_req zone=seckill burst=200 nodelay;
    limit_conn addr 10;
    
    location /api/seckill {
        proxy_pass http://gateway;
        proxy_set_header Host \$host;
        proxy_set_header X-Real-IP \$remote_addr;
        proxy_set_header X-Forwarded-For \$proxy_add_x_forwarded_for;
        proxy_http_version 1.1;
        proxy_set_header Connection "";
        
        # 超时配置
        proxy_connect_timeout 5s;
        proxy_send_timeout 10s;
        proxy_read_timeout 10s;
    }
    
    # 健康检查
    location /health {
        access_log off;
        return 200 "OK";
    }
}
EOF

# 3. 启动 Nginx
systemctl start nginx
systemctl enable nginx

# 4. 验证
curl http://localhost/health
```
